Problem 3.60 Solution:

long loop(long x, int n)
{
	long result = 0;
	long mask;
	for(mask = 1; mask != 0; mask = mask << n)
	{
		result |= (x & mask);
	}
	return result;
}


Problem 3.63 Solution:

long switch_prob(long x, long n)
{
	long result = x;
	switch(n)
	{
		case 62:
		case 60:
			result <<= 3;
			break;
		case 63:
			result >>= 3;
			break;
		case 64:
			result *= 15;
		case 65:
			result *= result;
		default:
			result += 75;
	}
	
	return result;
}
